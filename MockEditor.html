语法 Syntax

Type Syntax
String

'name|min-max': 'value'

'name|count': 'value'
Number

'name|+1': 100

'name|1-100': 100

'name|1-100.1-10
Boolean

'name|1': value

'name|min-max': value
Object

'name|min-max': {}

'name|count': {}
Array

'name|1': [{}, {} ...]

'name|min-max': [{}, {} ...]

'name|count': [{}, {} ...]
Function

'name': function(){}
